on_actions = {
    on_weekly_RUI = {
        effect = { 
            if = {
                limit = {
                    NOT = {
                        1017 = {
                            is_fully_controlled_by = RUI
                        }
                    }
                    has_idea = RUI_Rent_of_Yekaterinodar_port
                }
                remove_ideas = RUI_Rent_of_Yekaterinodar_port
            }
            else_if = {
                limit = {
                    1017 = {
                        is_fully_controlled_by = RUI
                    }
                    NOT = {
                        has_idea = RUI_Rent_of_Yekaterinodar_port
                    }
                }
                add_ideas = RUI_Rent_of_Yekaterinodar_port
            }
        }
    }
    on_startup = {
        effect = {
            RUI = {
                add_to_variable = {
                    RUI.production_of_artillery_shells_per_day = 10
                }
            }
        }
    }
    on_daily_RUI = {
        effect = {
            add_to_variable = { 
                RUI.artillery_shells_in_warehouses = RUI.production_of_artillery_shells_per_day 
            }
        }
    }
}
